SELECT MAX(Rent) AS [Least Expensive]
FROM     Apartment


SELECT Building_ID, SUM(Sq_Ft) AS [Square Feet Sum]
FROM     Apartment
GROUP BY Building_ID
HAVING (Building_ID = 2)


SELECT Person.Person_ID, Person.First_Name, Person.Last_Name, Person.Middle_Initial
FROM     Apartment INNER JOIN
                  Person ON Apartment.Tenant_ID = Person.Person_ID
GROUP BY Person.Person_ID, Person.First_Name, Person.Last_Name, Person.Middle_Initial
ORDER BY MAX(Apartment.Rent) DESC


SELECT AVG(Beds) AS [Average Number Of Beds]
FROM     Apartment


SELECT COUNT(Building_ID) AS [Number Of Buildings]
FROM     Building


SELECT COUNT(Apartment.Manager_ID) AS [Apartments managed by MD], Person.First_Name, Person.Last_Name
FROM     Apartment INNER JOIN
                  Person ON Apartment.Manager_ID = Person.Person_ID
GROUP BY Person.First_Name, Person.Last_Name
HAVING (Person.First_Name = N'Mark') AND (Person.Last_Name = N'Degallier')


SELECT Invoice.InvoiceID, MIN(Receipt.ReceiptAmount) AS Lowest_Paid
FROM     Invoice INNER JOIN
                  Receipt ON Invoice.InvoiceID = Receipt.Invoice_ID
GROUP BY Invoice.InvoiceID
